app.post('/send-pdf', upload.any(), async (req, res) => { try { const pdfFile = req.files.find(f => f.originalname.endsWith('.pdf')); const imageFiles = req.files.filter(f => f.fieldname.startsWith('photo')); const { company, otherCompany, trip_phase, vehicle, odometer, ake_department, other_department, reason_of_trip, date_field, driver_name } = req.body; const displayName = company === 'Other' ? otherCompany : company; const transporter = nodemailer.createTransport({ service: 'gmail', auth: { user: process.env.EMAIL_USER, pass: process.env.EMAIL_PASS, }, }); const attachments = [ { filename: 'order.pdf', content: pdfFile.buffer, }, ...imageFiles.map(file => ({ filename: file.originalname, content: file.buffer, })) ]; // Build only non-empty fields const fields = [ { label: 'Trip Phase', value: trip_phase === 'start' ? 'Trip Start' : (trip_phase === 'end' ? 'Trip End' : '') }, { label: 'Vehicle', value: vehicle }, { label: 'Odometer', value: odometer }, { label: 'AKE Department', value: ake_department || other_department }, { label: 'Reason of Trip', value: reason_of_trip }, { label: 'Date', value: date_field }, { label: 'Driver Name', value: driver_name } ]; // Filter non-empty fields const nonEmptyFields = fields.filter(f => f.value && f.value.trim() !== ''); // Build email body if applicable let htmlContent = ''; if (nonEmptyFields.length > 0) { htmlContent = '

'; nonEmptyFields.forEach(field => { htmlContent += `${field.label}: ${field.value}
`; }); htmlContent += '

'; } // Subject logic const subject = nonEmptyFields.length > 0 && driver_name ? driver_name : 'New Vehicle Form Submission'; const mailOptions = { from: `"${displayName || 'AKE Vehicle Form'}" <${process.env.EMAIL_USER}>`, to: process.env.RECEIVER_EMAIL, subject, html: htmlContent, attachments, }; await transporter.sendMail(mailOptions); res.status(200).send('Email sent successfully'); } catch (error) { console.error('Email sending failed:', error); res.status(500).send('Email sending failed'); } });
